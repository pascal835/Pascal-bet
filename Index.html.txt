
<script>
    // State management for selected bets
    let selectedBets = [];
    let stake = 1000; // Default stake amount

    // DOM Elements
    const oddButtons = document.querySelectorAll('.odd-btn');
    const betslipContainer = document.querySelector('.betslip');

    // Initialize Event Listeners
    oddButtons.forEach(button => {
        button.addEventListener('click', () => handleOddSelection(button));
    });

    // Handle when an odds button is clicked
    function handleOddSelection(button) {
        const matchCard = button.closest('.match-card');
        const matchId = matchCard.querySelector('.teams').textContent.trim();
        const teams = matchCard.querySelector('.teams').textContent.trim();
        const market = button.querySelector('.odd-label').textContent.trim();
        const odds = parseFloat(button.querySelector('.odd-value').textContent.trim());

        // Create a unique ID combining match and market
        const betId = `${matchId}-${market}`;

        // Check if this exact bet is already selected
        const existingBetIndex = selectedBets.findIndex(bet => bet.id === betId);

        if (existingBetIndex > -1) {
            // Deselect: Remove from array and remove active class
            selectedBets.splice(existingBetIndex, 1);
            button.style.backgroundColor = '';
            button.style.color = '';
            button.querySelector('.odd-label').style.color = '';
        } else {
            // Optional: If you only want ONE selection per match, filter out existing match selections here.
            // For this layout, we'll allow multiple or simple accumulator logic.
            
            // Select: Add to array and style button active
            selectedBets.push({ id: betId, matchId, teams, market, odds });
            button.style.backgroundColor = 'var(--primary-color)';
            button.style.color = 'var(--text-light)';
            button.querySelector('.odd-label').style.color = '#ffd7d7';
        }

        renderBetslip();
    }

    // Calculate total odds (Accumulator style: multiplying the odds together)
    function calculateTotalOdds() {
        if (selectedBets.length === 0) return 0;
        return selectedBets.reduce((total, bet) => total * bet.odds, 1).toFixed(2);
    }

    // Render the updated betslip DOM
    function renderBetslip() {
        // If no bets are selected, show the empty state
        if (selectedBets.length === 0) {
            betslipContainer.innerHTML = `
                <h3>Bet Slip</h3>
                <div class="empty-betslip">
                    <p>Your bet slip is empty.</p>
                    <p style="font-size: 12px; margin-top: 10px;">Click on odds to add them to your bet slip.</p>
                </div>
            `;
            return;
        }

        const totalOdds = calculateTotalOdds();
        const potentialPayout = (totalOdds * stake).toFixed(2);

        // Generate HTML for selected items
        let betsHtml = selectedBets.map(bet => `
            <div class="betslip-item" style="padding: 10px 0; border-bottom: 1px solid var(--border-color); position: relative;">
                <div style="font-size: 12px; color: #666; margin-bottom: 4px;">${bet.teams}</div>
                <div style="display: flex; justify-content: space-between; font-weight: bold; font-size: 14px;">
                    <span>Market: ${bet.market}</span>
                    <span style="color: var(--primary-color);">${bet.odds}</span>
                </div>
            </div>
        `).join('');

        // Inject the active betslip interface
        betslipContainer.innerHTML = `
            <h3>Bet Slip (${selectedBets.length})</h3>
            <div class="betslip-items-list" style="max-height: 300px; overflow-y: auto;">
                ${betsHtml}
            </div>
            
            <div class="betslip-summary" style="margin-top: 20px; border-top: 2px solid var(--border-color); padding-top: 15px;">
                <div style="display: flex; justify-content: space-between; margin-bottom: 10px; font-weight: bold;">
                    <span>Total Odds:</span>
                    <span style="color: var(--primary-color); font-size: 18px;">${totalOdds}</span>
                </div>
                
                <div style="margin-bottom: 15px;">
                    <label style="display: block; font-size: 12px; color: #666; margin-bottom: 5px;">Stake Amount</label>
                    <input type="number" id="stake-input" value="${stake}" style="width: 100%; padding: 8px; border: 1px solid var(--border-color); border-radius: 4px; font-weight: bold; font-size: 16px;">
                </div>

                <div style="display: flex; justify-content: space-between; margin-bottom: 15px; font-weight: bold; background: #f9f9f9; padding: 10px; border-radius: 4px;">
                    <span>Potential Winnings:</span>
                    <span style="color: green;">₦${Number(potentialPayout).toLocaleString()}</span>
                </div>

                <button style="width: 100%; background-color: var(--primary-color); color: white; border: none; padding: 12px; border-radius: 4px; font-weight: bold; font-size: 16px; cursor: pointer; transition: opacity 0.2s;">
                    Place Bet
                </button>
            </div>
        `;

        // Re-attach listener to the newly rendered stake input
        const stakeInput = document.getElementById('stake-input');
        stakeInput.addEventListener('input', (e) => {
            stake = parseFloat(e.target.value) || 0;
            // Update potential winnings display in real-time without rerendering entire component
            const winningsSpan = betslipContainer.querySelector('.betslip-summary span style*="color: green"');
            const updatedPayout = (calculateTotalOdds() * stake).toFixed(2);
            betslipContainer.querySelector('span[style*="color: green"]').textContent = `₦${Number(updatedPayout).toLocaleString()}`;
        });
    }
</script>
